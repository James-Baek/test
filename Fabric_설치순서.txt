5. golang
	$ wget https://dl.google.com/go/go1.10.linux-amd64.tar.gz
	$ sudo tar -xvf go1.10.linux-amd64.tar.gz
	$ sudo mv go /usr/local
	$ export GOROOT=/usr/local/go
	$ export PATH=$PATH:$GOROOT/bin:
	$ go version (1.10 확인)
