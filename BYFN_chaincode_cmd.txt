# golang
peer chaincode install -n mycc_node -v 1.0 -p github.com/chaincode/chaincode_example02/node

# nodejs
peer chaincode install -n mycc_node -l node -v 1.0 -p /opt/gopath/src/github.com/chaincode/chaincode_example02/node

# java
peer chaincode install -n mycc_java -l java -v 1.0 -p /opt/gopath/src/github.com/chaincode/chaincode_example02/java

# create instance for golang chaincode
